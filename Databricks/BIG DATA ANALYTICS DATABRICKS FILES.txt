graph = nx.DiGraph()
nodes = ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'I']
edges = [
    ('A', 'B'), ('B', 'C'), ('C', 'D'), ('D', 'E'), ('E', 'F'),
    ('F', 'G'), ('G', 'H'), ('H', 'I'), ('I', 'A'), ('A', 'C'),
    ('C', 'E'), ('E', 'G')
]
graph.add_nodes_from(nodes)
graph.add_edges_from(edges)
nx.draw_networkx(graph)
nx.in_degree_centrality(graph)
nx.out_degree_centrality(graph)
nx.shortest_path(graph,'A')
list(nx.strongly_connected_components(graph))

nx.density(graph)

degree_sequence = [d for _,d in nx.degree(graph)]
plt.hist(degree_sequence, bins=range(1, max(degree_sequence) + 2))
plt.show()
nx.average_shortest_path_length(graph)
